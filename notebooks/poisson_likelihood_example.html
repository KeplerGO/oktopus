%matplotlib inline
import oktopus
from oktopus import UniformPrior, JointPrior, PoissonPosterior, PoissonLikelihood
import numpy as np
import matplotlib.pyplot as pl

In [2]:

x = np.linspace(0, 5, 100)
def mean(m, b):
    return x * m + b
poisson_data = np.random.poisson(mean(2, 10))

In [3]:

pl.step(x, poisson_data)
